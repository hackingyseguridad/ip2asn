# asn2ip
Utilidad para extraer las IP correspondientes a un AS

# ip2asn
Utilidad para extraer el AS de una IP

# ip2pais
Utilidad para ver el pais de una IP

# traceasn
traceroute con el nombre extendido de la operadora del AS en cada salto 

# Instalacion
sh instalar.sh

# Uso:

asn2ip

ip2asn

ip2pais

traceasn
